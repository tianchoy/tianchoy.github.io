    父组件向子组件传参


    
//在父级组件中使用NewsList 而
<el-main>
    <NewsList :newsData="newsData" />
</el-main>


//父级组件中引用子级组件
import NewsList from '../components/NewsList.vue'
import {reactive,toRefs} from 'vue'
export default {
    components:{
        NewsList
    },

    setup(){
        const data = reactive({
            newsData:[]
        })

        //请求接口，省略步请求步骤
        state.newsData = res.data.data

        return {
            ...toRefs(data)
        }
    }
}



    //子组件(NewsList.vue)中应用数据
<div class="newsList">
    <div v-for="item in newsData" :key="item.id">
        <h4>{{item.id}}</h4>
    </div>
</div>
//子级组件接受数据
<script lang="ts">
    export default {
        props:{
            newsData:{
                type:Array,
                default:[]
            }
        }
    }
</script>

    子组件向父组件传参，用分页功能举例


    //子组件 Pag.vue
    <template>
        <el-config-provider>
            <el-pagination
                @current-change="handleCurrentChange"
                >
            </el-pagination>
        </el-config-provider>
    </template>

    import { reactive, toRefs } from 'vue';
    export default({
        setup(props:any,{emit}:any) {
            const handleCurrentChange = (page:any) => {
                emit('pageNumber',page) // 用emit 传参
            };

            return {
                handleCurrentChange,
            };
        },
    })


    //父组件接受子组件里page
<Pagination @pageNumber='pageNumber' />


import Pagination from '../components/Pag.vue'
